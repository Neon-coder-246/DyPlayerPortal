import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'; export default async (req) => { const supabase = createClient( Deno.env.get("SUPABASE_URL"), Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ); const url = new URL(req.url); const macParam = url.searchParams.get("mac"); const path = url.pathname; // ---------------------------------------------------- // 1. UI MODE (no ?mac= param) // ---------------------------------------------------- if (req.method === "GET" && !macParam) { const html = ` DyPlayer Portal

Update Playlist

`; return new Response(html, { status: 200, headers: { "Content-Type": "text/html" } }); } // ---------------------------------------------------- // 2. API MODE (app requests playlist) // ---------------------------------------------------- if (req.method === "GET" && macParam) { const { data, error } = await supabase .from("playlists") .select("playlist") .eq("mac", macParam) .single(); if (error || !data) { return new Response("MAC not found", { status: 404 }); } return new Response(data.playlist, { status: 200, headers: { "Content-Type": "text/plain" } }); } // ---------------------------------------------------- // 3. ADMIN MODE (update playlist) // ---------------------------------------------------- if (req.method === "POST" && path === "/admin") { const body = await req.json(); const mac = (body.mac || "").trim(); const playlist = (body.playlist || "").trim(); if (!mac || !playlist) { return new Response(JSON.stringify({ error: "mac and playlist required" }), { status: 400, headers: { "Content-Type": "application/json" } }); } const { error } = await supabase .from("playlists") .upsert({ mac, playlist }); if (error) { return new Response(JSON.stringify({ error: error.message }), { status: 500, headers: { "Content-Type": "application/json" } }); } return new Response(JSON.stringify({ ok: true }), { status: 200, headers: { "Content-Type": "application/json" } }); } return new Response("Not found", { status: 404 }); };
